-- a) 
SELECT ds_nome, ds_logradouro
FROM Proprietario;

-- b)
SELECT *
FROM modelo
WHERE cd_marca = 4
ORDER BY ds_modelo;

-- c) 
SELECT ds_marca, ds_modelo
FROM modelo v INNER JOIN Marca mar
ORDER BY mar.ds_marca Asc, ds_modelo;

-- d)
SELECT nr_placa, nr_ano_mod, ds_cor
FROM Veiculo v INNER JOIN Cor 
ORDER BY nr_ano_mod, ds_cor;

-- e)
SELECT nr_placa, nm_proprietario, ds_logradouro
FROM Veiculo v INNER JOIN Proprietario p
WHERE p.sg_uf = "SC"

-- f)
SELECT nr_placa, ds_marca, ds_modelo, nr_ano_mod, ds_combustivel
FROM Veiculo v INNER JOIN Modelo, Veiculo_Combustivel, Combustivel, Marca

-- g)
SELECT nr_placa, ds_marca, ds_modelo, ds_acessorio
FROM Veiculo V INNER JOIN Modelo, Marca, Veiculo_Acessorio, Acessorio
WHERE ds_acessorio = "Direção Hidráulica"

-- h)
SELECT *
FROM Veiculo V INNER JOIN Modelo, Marca, Veiculo_Acessorio, Acessorio, Veiculo_Combustivel, Combustivel
